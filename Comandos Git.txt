### Lista de comandos Git

ls -a = arquivos ocultos 

ctrl+L = Limpar tela

git commit =gravar modificações

git commit -m =descrição da ataualização do projeto

git clone =clonar arquivos

git add =adicionar arquivos ao stage

git push =subir modificação para o repositório remoto

git status =ver quais arquivos estão sendo rastreados

git branch =ver todas branch presente no repositório

git branch -r = branch repositório remoto

git checkout = trocar de branch

git config = configurar sistema git

git init = para criar um novo projeto

git remote = conexão entre seu repositório local e um repos. remoto

git rm = remover arquivos da pasta

git help =ajuda diretamente no terminal
